# Gamification + Ideas Board Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a public Ideas Board with voting, comments, media, and a full gamification layer (badges, Hall of Fame, user profile pages) to the school-committee Next.js app.

**Architecture:** Extend the existing `Idea`/`IdeaVote` Prisma models; add `IdeaPhoto`, `IdeaVideo`, `IdeaComment`, `IdeaCommentLike`, `Achievement`, and `UserAchievement` tables. A synchronous badge-award engine (`lib/gamification/award-badges.ts`) is called at the end of every relevant mutation. New pages live under `app/(public)/ideas/` (public read) and `app/(app)/` (auth-only: Hall of Fame, profiles). Existing `/feedback` is untouched.

**Tech Stack:** Next.js 14 App Router, TypeScript strict, Prisma ORM (PostgreSQL at 192.168.88.53:5432), Tailwind CSS, MinIO (S3-compatible) for media, existing `lib/auth/get-current-user.ts` for auth, existing `lib/db/audit.ts` for audit events.

import { db } from "@/lib/db/client";
import { writeAuditEvent } from "@/lib/db/audit";
import { buildPage, resolveLimit, type PageParams, type PageResult } from "@/lib/db/pagination";
import { AppError, NotFoundError } from "@/types/errors";

function parseFileKey(fileKey: string): { fileId: string; fileExt: string } {
  const filename = fileKey.split("/").pop() ?? fileKey;
  const dotIdx = filename.lastIndexOf(".");
  return dotIdx >= 0
    ? { fileId: filename.slice(0, dotIdx), fileExt: filename.slice(dotIdx + 1) }
    : { fileId: filename, fileExt: "bin" };
}

export interface CreateIdeaParams {
  schoolId: string;
  classId?: string;
  submittedBy: string;
  title: string;
  description: string;
  isAnonymous: boolean;
  voiceFileKey?: string;
  voiceTranscript?: string;
  photoFileKeys?: string[];
  videoFileKeys?: string[];
  tenantId: string;
  requestId?: string;
}

export interface IdeaWithCounts {
  id: string;
  schoolId: string;
  title: string;
  description: string;
  isAnonymous: boolean;
  status: string;
  categories: string[];
  voiceFileKey: string | null;
  voiceTranscript: string | null;
  createdAt: Date;
  updatedAt: Date;
  submittedBy: string | null;
  voteCount: number;
  commentCount: number;
  photos: { fileId: string; fileExt: string; sortOrder: number }[];
  videos: { fileId: string; fileExt: string; sortOrder: number }[];
}

export async function createIdea(params: CreateIdeaParams): Promise<{ id: string }> {
  return db.$transaction(async (tx) => {
    const idea = await tx.idea.create({
      data: {
        schoolId: params.schoolId,
        classId: params.classId ?? null,
        submittedBy: params.submittedBy,
        title: params.title,
        description: params.description,
        isAnonymous: params.isAnonymous,
        voiceFileKey: params.voiceFileKey ?? null,
        voiceTranscript: params.voiceTranscript ?? null,
        status: "active",
      },
    });

    if (params.photoFileKeys?.length) {
      await tx.ideaPhoto.createMany({
        data: params.photoFileKeys.map((key, i) => {
          const { fileId, fileExt } = parseFileKey(key);
          return { ideaId: idea.id, fileId, fileExt, sortOrder: i };
        }),
      });
    }

    if (params.videoFileKeys?.length) {
      await tx.ideaVideo.createMany({
        data: params.videoFileKeys.map((key, i) => {
          const { fileId, fileExt } = parseFileKey(key);
          return { ideaId: idea.id, fileId, fileExt, sortOrder: i };
        }),
      });
    }

    await writeAuditEvent(
      {
        tenantId: params.tenantId,
        schoolId: params.schoolId,
        actorUserId: params.submittedBy,
        action: "idea.created",
        entityType: "idea",
        entityId: idea.id,
        requestId: params.requestId,
      },
      tx,
    );

    return { id: idea.id };
  });
}

export async function listIdeas(
  schoolId: string,
  params?: PageParams,
): Promise<PageResult<IdeaWithCounts>> {
  const limit = resolveLimit(params?.limit);
  const rows = await db.idea.findMany({
    where: { schoolId, status: "active" },
    orderBy: { votes: { _count: "desc" } },
    take: limit + 1,
    ...(params?.cursor ? { cursor: { id: params.cursor }, skip: 1 } : {}),
    include: {
      _count: { select: { votes: true, comments: true } },
      photos: { orderBy: { sortOrder: "asc" }, select: { fileId: true, fileExt: true, sortOrder: true } },
      videos: { orderBy: { sortOrder: "asc" }, select: { fileId: true, fileExt: true, sortOrder: true } },
    },
  });

  const mapped = rows.map((r) => ({
    id: r.id,
    schoolId: r.schoolId,
    title: r.title,
    description: r.description,
    isAnonymous: r.isAnonymous,
    status: r.status,
    categories: r.categories,
    voiceFileKey: r.voiceFileKey,
    voiceTranscript: r.voiceTranscript,
    createdAt: r.createdAt,
    updatedAt: r.updatedAt,
    submittedBy: r.submittedBy,
    voteCount: r._count.votes,
    commentCount: r._count.comments,
    photos: r.photos,
    videos: r.videos,
  }));

  return buildPage(mapped, limit);
}

export async function getIdeaById(id: string): Promise<IdeaWithCounts> {
  const r = await db.idea.findUnique({
    where: { id },
    include: {
      _count: { select: { votes: true, comments: true } },
      photos: { orderBy: { sortOrder: "asc" }, select: { fileId: true, fileExt: true, sortOrder: true } },
      videos: { orderBy: { sortOrder: "asc" }, select: { fileId: true, fileExt: true, sortOrder: true } },
    },
  });
  if (!r || r.status === "deleted") throw new NotFoundError("Idea not found");
  return {
    id: r.id,
    schoolId: r.schoolId,
    title: r.title,
    description: r.description,
    isAnonymous: r.isAnonymous,
    status: r.status,
    categories: r.categories,
    voiceFileKey: r.voiceFileKey,
    voiceTranscript: r.voiceTranscript,
    createdAt: r.createdAt,
    updatedAt: r.updatedAt,
    submittedBy: r.submittedBy,
    voteCount: r._count.votes,
    commentCount: r._count.comments,
    photos: r.photos,
    videos: r.videos,
  };
}

export async function softDeleteIdea(
  id: string,
  actorUserId: string,
  tenantId: string,
  schoolId: string,
  requestId?: string,
): Promise<void> {
  const idea = await db.idea.findUnique({ where: { id } });
  if (!idea) throw new NotFoundError("Idea not found");

  await db.$transaction(async (tx) => {
    await tx.idea.update({ where: { id }, data: { status: "deleted" } });
    await writeAuditEvent(
      { tenantId, schoolId, actorUserId, action: "idea.deleted", entityType: "idea", entityId: id, requestId },
      tx,
    );
  });
}

export async function toggleIdeaVote(
  ideaId: string,
  userId: string,
  tenantId: string,
  schoolId: string,
  requestId?: string,
): Promise<{ voted: boolean; voteCount: number }> {
  const idea = await db.idea.findUnique({ where: { id: ideaId } });
  if (!idea || idea.status !== "active") throw new NotFoundError("Idea not found");
  if (idea.submittedBy === userId) {
    throw new AppError("FORBIDDEN", "Cannot vote on your own idea", 403);
  }

  const existing = await db.ideaVote.findUnique({
    where: { ideaId_userId: { ideaId, userId } },
  });

  await db.$transaction(async (tx) => {
    if (existing) {
      await tx.ideaVote.delete({ where: { ideaId_userId: { ideaId, userId } } });
      await writeAuditEvent(
        { tenantId, schoolId, actorUserId: userId, action: "idea.vote_removed", entityType: "idea", entityId: ideaId, requestId },
        tx,
      );
    } else {
      await tx.ideaVote.create({ data: { ideaId, userId, voteType: "support" } });
      await writeAuditEvent(
        { tenantId, schoolId, actorUserId: userId, action: "idea.voted", entityType: "idea", entityId: ideaId, requestId },
        tx,
      );
    }
  });

  const voteCount = await db.ideaVote.count({ where: { ideaId } });
  return { voted: !existing, voteCount };
}

export async function getUserVotedIdeaIds(userId: string, ideaIds: string[]): Promise<Set<string>> {
  const votes = await db.ideaVote.findMany({
    where: { userId, ideaId: { in: ideaIds } },
    select: { ideaId: true },
  });
  return new Set(votes.map((v) => v.ideaId));
}

export async function awardBadgesForUser(userId: string): Promise<void> {
  const [
    ideaCount,
    voteCount,
    commentCount,
    taskCompletedCount,
    profile,
    children,
    roles,
    maxIdeaVotes,
    hasRegistered,
  ] = await Promise.all([
    db.idea.count({ where: { submittedBy: userId, status: "active" } }),
    db.ideaVote.count({ where: { userId } }),
    db.ideaComment.count({ where: { userId } }),
    db.taskStatusEvent.count({ where: { actorUserId: userId, newStatus: "completed" } }),
    db.profile.findUnique({ where: { userId }, select: { firstName: true, lastName: true, phone: true, createdAt: true } }),
    db.child.findMany({ where: { parentUserId: userId }, select: { id: true } }),
    db.userRole.findMany({ where: { userId, revokedAt: null }, select: { role: true } }),
    db.ideaVote.groupBy({
      by: ["ideaId"],
      where: { idea: { submittedBy: userId, status: "active" } },
      _count: { ideaId: true },
      orderBy: { _count: { ideaId: "desc" } },
      take: 1,
    }),
    db.profile.findUnique({ where: { userId }, select: { userId: true } }),
  ]);

  const badges: Array<() => Promise<void>> = [];

  // Registration
  if (hasRegistered) badges.push(() => awardAchievement(userId, "registered"));

  // Profile complete: name + phone + at least 1 child
  if (profile?.firstName && profile.lastName && profile.phone && children.length > 0) {
    badges.push(() => awardAchievement(userId, "profile_complete"));
  }

  // Child added
  if (children.length > 0) badges.push(() => awardAchievement(userId, "child_added"));

  // Role badges
  const roleSet = new Set(roles.map((r) => r.role));
  if (roleSet.has("committee")) badges.push(() => awardAchievement(userId, "role_committee"));
  if (roleSet.has("teacher")) badges.push(() => awardAchievement(userId, "role_teacher"));

  // Idea badges
  badges.push(
    () => checkThreshold(userId, "idea_first", ideaCount, 1),
    () => checkThreshold(userId, "idea_5", ideaCount, 5),
    () => checkThreshold(userId, "idea_20", ideaCount, 20),
  );

  // Vote badges
  badges.push(
    () => checkThreshold(userId, "voter_first", voteCount, 1),
    () => checkThreshold(userId, "voter_10", voteCount, 10),
    () => checkThreshold(userId, "voter_50", voteCount, 50),
  );

  // Comment badges
  badges.push(
    () => checkThreshold(userId, "comment_first", commentCount, 1),
    () => checkThreshold(userId, "comment_10", commentCount, 10),
  );

  // Popular idea badges (max votes on any single idea by this user)
  const topVoteCount = maxIdeaVotes[0]?._count?.ideaId ?? 0;
  badges.push(
    () => checkThreshold(userId, "popular_idea_10", topVoteCount, 10),
    () => checkThreshold(userId, "popular_idea_50", topVoteCount, 50),
  );

  // Task badges
  badges.push(
    () => checkThreshold(userId, "task_completed", taskCompletedCount, 1),
    () => checkThreshold(userId, "task_champion_5", taskCompletedCount, 5),
    () => checkThreshold(userId, "task_champion_20", taskCompletedCount, 20),
  );

  // Early adopter: registered within first 30 days of Sept 1 of that year
  if (profile?.createdAt) {
    const created = profile.createdAt;
    const sept1 = new Date(created.getFullYear(), 8, 1); // month 8 = September
    const diff = created.getTime() - sept1.getTime();
    if (diff >= 0 && diff <= 30 * 24 * 60 * 60 * 1000) {
      badges.push(() => awardAchievement(userId, "early_adopter"));
    }
  }

  await Promise.all(badges.map((fn) => fn()));
}

export async function recomputeBestCommenter(): Promise<void> {
  const topUserId = await getTopUserByCommentLikes();

  // Find current holder
  const current = await db.userAchievement.findFirst({
    where: { achievementKey: "best_commenter" },
  });

  if (!topUserId) return;

  if (current && current.userId !== topUserId) {
    await revokeAchievement(current.userId, "best_commenter");
    await awardAchievement(topUserId, "best_commenter");
  } else if (!current) {
    await awardAchievement(topUserId, "best_commenter");
  }
}

  const requestId = getOrCreateRequestId(req.headers.get("x-request-id"));
  try {
    await getCurrentUser(requestId); // auth required

    const tierMap = await getTierMap();

    // Get all users with at least one achievement, group by userId
    const rows = await db.userAchievement.findMany({
      select: { userId: true, achievementKey: true },
    });

    // Group achievements by userId
    const byUser = new Map<string, string[]>();
    for (const row of rows) {
      if (!byUser.has(row.userId)) byUser.set(row.userId, []);
      byUser.get(row.userId)!.push(row.achievementKey);
    }

    // Calculate scores
    const scored = Array.from(byUser.entries()).map(([userId, keys]) => ({
      userId,
      score: calcUserScore(keys.map((k) => ({ achievementKey: k, tier: tierMap[k] })), tierMap),
      achievementKeys: keys,
    }));

    // Sort by score desc, take top 20
    scored.sort((a, b) => b.score - a.score);
    const top20 = scored.slice(0, 20);

    // Fetch profiles for top20
    const userIds = top20.map((u) => u.userId);
    const profiles = await db.profile.findMany({
      where: { userId: { in: userIds } },
      select: { userId: true, firstName: true, lastName: true },
    });
    const profileMap = new Map(profiles.map((p) => [p.userId, p]));

    const result = top20.map((u, i) => {
      const profile = profileMap.get(u.userId);
      return {
        rank: i + 1,
        userId: u.userId,
        firstName: profile?.firstName ?? "",
        lastName: profile?.lastName ?? "",
        score: u.score,
        achievementKeys: u.achievementKeys,
      };
    });

    return NextResponse.json({ items: result }, { status: 200 });
  } catch (err) {
    if (err instanceof AppError) {
      return NextResponse.json(toErrorResponse(err, requestId), { status: err.statusCode });
    }
    logger.error("hall-of-fame GET: unexpected", { request_id: requestId, route: ROUTE, error_message: err instanceof Error ? err.message : String(err) });
    return NextResponse.json(toErrorResponse(new AppError("INTERNAL_ERROR", "Unexpected error", 500), requestId), { status: 500 });
  }
}

import { tryGetCurrentUser } from "@/lib/auth/get-current-user";
import { getOrCreateRequestId } from "@/lib/request-id";

export async function GET(req: NextRequest) {
  const requestId = getOrCreateRequestId(req.headers.get("x-request-id"));
